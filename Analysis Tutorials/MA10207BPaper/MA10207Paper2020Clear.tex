\BeginKnitrBlock{Question*}
{}Let \(f:\mathbb{R}\to\mathbb{R}\) be given by \[f(x) = \frac{2x^2 - 2}{x^2 + x + 2}.\]

\begin{enumerate}
\def\labelenumi{\alph{enumi})}
\tightlist
\item
  Using the \(\epsilon-\delta\) definition, prove that \[\lim_{x\to 0}f(x) = -1.\]
\item
  Using the \emph{Algebra of Limits} and the \emph{Sequential Characterisation of Limits}, show that \[\lim_{x \to \pm 1} f(x) = 0.\]
\end{enumerate}
\EndKnitrBlock{Question*}

\BeginKnitrBlock{solution*}
\begin{enumerate}
\def\labelenumi{\alph{enumi})}
\tightlist
\item
  Fix \(\epsilon > 0\) and suppose \(0 < \lvert x - 0 \rvert < \delta\) for some \(\delta > 0\) to be chosen later. Then
  \begin{align*}
  \lvert f(x) - (-1)\rvert &= \left\lvert \frac{2x^2 - 2}{x^2 + x + 2} - (-1) \right\rvert,\\
  &= \frac{\lvert x \rvert\lvert 3x + 1\rvert}{\lvert x^2 + x + 2\rvert}.
  \end{align*}
  Now,
  \begin{align*}
  \lvert 3x + 1\rvert &\leq 3\lvert x \rvert + 1 \;\;&&\text{(by the triangle ineq.)},\\
  &< 4 \;\; &&\text{(if $\delta < 1$)}.
  \end{align*}
\end{enumerate}
\EndKnitrBlock{solution*}
